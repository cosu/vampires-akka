package ro.cosu.vampires.server.values.resources;

import java.util.Iterator;
import java.util.Map;
import java.util.Spliterator;
import java.util.Spliterators;
import java.util.function.Consumer;

public class ConfigurationIterable implements Iterable<Configuration> {

    private final Map<ResourceDescription, Integer> maxResources;

    public ConfigurationIterable(Map<ResourceDescription, Integer> maxResources) {
        this.maxResources = maxResources;
    }

    @Override
    public Iterator<Configuration> iterator() {
        return new ConfigurationIterator(maxResources);
    }

    @Override
    public void forEach(Consumer<? super Configuration> action) {
        iterator().forEachRemaining(action);
    }

    @Override
    public Spliterator<Configuration> spliterator() {
        Integer size = maxResources.values().stream().reduce(1, (a, b) -> a * b);
        return Spliterators.spliterator(this.iterator(), size, 0);
    }
}
